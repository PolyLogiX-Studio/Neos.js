Example
const NEOS = require("@bombitmanbomb/neosjs");
const Neos = new NEOS();
const { CommandManager } = require("neosjs-commands")
const Command = CommandManager.CreateCommands(Neos));
Neos.on("error", (err)=>throw new Error(err));
Neos.on("messageReceived", (Message)=>{
	switch(Message.MessageType){
	case "Text":
		Command.Run(Message)
	break
	default:
		Neos.SendTextMessage(Message.SenderId,"Message could not be handled.");
	}
})
Command.Add("ping", (Handler)=>{
	Handler.Reply("Pong!");
},
{
	index:"Ping Pong!"
	usage: Command.Options.Prefix + "ping"
});

new Neos(options: NeosGeneratorOptions?)

Extends EventEmitter

Parameters
options (NeosGeneratorOptions?)
Name Description
options.AutoReadMessages boolean (default true) Auto Mark messages as Read in the API when handled by the Neos#messageReceived event
options.OnlineState OnlineStatus (default "Online") Online User Status, Displayed in User List.
options.StatusInterval number (default 60) How Frequent to update the user status. Will trigger Neos#statusUpdated .
options.NeosVersion string (default "Neos.js ${PackageVersion}") Defaults to the Library version. Use to display a custom "Neos Version"
options.CompatabilityHash string (default "Neos.js ${PackageVersion}") Determines Version Compatability Hash, Used by clients to determing if a session is the same version.
options.UpdateInterval number (default 1000) Internal Update interval in ms . Reccomended: 1000
options.Update boolean (default true) Enable State of the Internal Update loop. Only reccomend setting this to false if you know what you are doing.
options.MultiMessageDelay number (default 1100) MultiMessage Delay for sending messages in ms . Reccomend over 1000ms
